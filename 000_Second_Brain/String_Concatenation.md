# String Concatenation

The simplest way to join string is using the + sign ie

'# Environment name'
environment = 'dev'

'# Application name'
application = 'appserver'

'# Instances Number'
instance_number = "02"

'# Concatenate'
instance_name = environment + '-' + application + '-instance-' + instance_number

'# Print'
print('EC2 Instance name: ' + instance_name)
